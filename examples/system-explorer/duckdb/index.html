Quick Start

Get up and running with DuckDB in seconds. No server, no configuration, no dependencies.

# Install DuckDB
pip install duckdb
import duckdb

# Query files directly -- no CREATE TABLE needed
duckdb.sql("SELECT * FROM 'sales_2025.parquet' WHERE region = 'APAC' LIMIT 10").show()

# Query pandas DataFrames with zero-copy
import pandas as pd
orders = pd.DataFrame({"product": ["Widget", "Gadget"], "revenue": [100, 200]})
duckdb.sql("SELECT product, SUM(revenue) FROM orders GROUP BY product").df()
